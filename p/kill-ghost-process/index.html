ALL_PIDS=$(fuser /dev/nvidia* 2>/dev/null | tr ' ' '\n' | sort -u)

for pid in $ALL_PIDS; do
  if [[ ! " ${KEEP_PIDS[@]} " =~ " ${pid} " ]]; then
    echo "Killing ghost process PID $pid"
    sudo kill -9 $pid
  else
    echo "Keeping important process PID $pid"
  fi
done
