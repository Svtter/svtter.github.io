        for(int i = 1; i < 10;i  ++)

                for(int k = ; k < 10; k++)
                {
                    if(((i+j)*2+k) == n)
                        printf("%d%d%d%d%d\n", i, j, k, j, i);
                }
        for(int i = 1; i < 10;i  ++)
