lln sum[maxn];
struct ele
{
    lln x, w;
};
int cmp(ele a, ele b)
{
    return a.x < b.x;
}
ele e[maxn];
int main()

        for(int i = ; i < n; i++) scanf("%lld", &e[i].w);
        sort(e, e+n, cmp);
        sum[] = e[].w;
        for(int i = 1; i < n;i ++) sum[i] = e[i].w + sum[i-1];
        double sm = sum[n-1]/2.0;
        bool flag;
        int i;
        for(i = ; i < n; i++)
        {
            int ti, tj;
            ti = tj = i;

            else ti = -1;
            if(i != n-1) while(e[tj].x == e[tj+1].x) tj++;
            else tj = n;
