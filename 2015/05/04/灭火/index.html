火会蔓延，人被火追着跑，能否跑出边界的问题。

bfs火之后bfs人，或火和人放在一个队列里面bfs

坑是没有火的情况，如果从0更新火势图而不是inf开始，那么可能造成没火不能跑的情况。

struct Node {
    int i, j, t;

void init(queue <Node> &q) {
    Node temp, next;
    int nx, ny;
    while(!q.empty()) {
        temp = q.front(), q.pop();
        temp.t++;
        for(int i = ; i < 4; i++) {
            nx = temp.i + dx[i];
            ny = temp.j + dy[i];
            if(g[nx][ny] != '#' && judge(nx, ny) && gex[nx][ny] > temp.t) {
                next.i = nx;
                next.j = ny;
                next.t = temp.t;
                gex[nx][ny] = temp.t;
                q.push(next);
            }
        }
    }
}
int bfs(queue <Node> &q) {
    Node t, n;
    int nx, ny;
    while(!q.empty()) {
        t = q.front(), q.pop();
        if(t.i == r-1 || t.j == c-1 || t.i ==  || t.j == ) return t.t;
        t.t ++;
        for(int i = ; i < 4; i++)
        {
            nx = t.i + dx[i];
            ny = t.j + dy[i];
            if(!vis[nx][ny] && gex[nx][ny] > t.t && judge(nx, ny))
            {
                n.i = nx, n.j = ny, n.t = t.t;
                vis[nx][ny] = 1;
                q.push(n);
            }
        }
    }

}
